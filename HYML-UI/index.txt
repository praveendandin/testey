        function buildNewsSlider() {
            $.get('apps/blog/post/tag/homeslider', function (postContent) {
                for (i = 0; i <= 2; i++) {
                    let excerptLength = 200;
                    let featureImg = $(postContent).find(".postHeader:eq(" + i + ")").find('img').attr('src');
                    let postTitle = $(postContent).find(".postHeader:eq(" + i + ")").find('.postTitle a').text();
                    let postUrl = $(postContent).find(".postHeader:eq(" + i + ")").find('.postTitle a').attr('href');
                    let postExOrg = $(postContent).find(".postContent:eq(" + i + ")").text().length;
                    let postEx = $(postContent).find(".postContent:eq(" + i + ")").text().substring(0, excerptLength);
                    postEx = postEx.replace(/\s+\S*$/, "");
                    let postCat = $(postContent).find(".postHeader:eq(" + i + ")").find('.postCategory a').text();

                    //Build Slides
                    $('.news-slide' + i).css('background-image', 'url(' + featureImg + ')');
                    $('.news-slide' + i + ' .news-card-title').append('<a href="' + postUrl + '">' + postTitle + '</a>');
                    if (postExOrg >= excerptLength) {
                        $('.news-slide' + i + ' .news-card-body').append(postEx + '...');
                    } else {
                        $('.news-slide' + i + ' .news-card-body').append(postEx);
                    }
                    $('.news-slide' + i + ' .news-card-tag').append('<a href="/apps/blog/post/category/' + postCat + '">' + postCat + '</a>');
                    $('.news-slide' + i + ' .news-card-btn').attr('href', postUrl);
                }
            });
        }
